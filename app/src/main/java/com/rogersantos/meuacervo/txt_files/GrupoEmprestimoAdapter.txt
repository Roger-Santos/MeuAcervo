package com.rogersantos.meuacervo.ui.controle

import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.ImageView
import android.widget.LinearLayout
import android.widget.TextView
import androidx.core.content.ContextCompat
import androidx.recyclerview.widget.DiffUtil
import androidx.recyclerview.widget.ListAdapter
import androidx.recyclerview.widget.RecyclerView
import com.bumptech.glide.Glide
import com.rogersantos.meuacervo.R
import java.text.SimpleDateFormat
import java.util.*

class GrupoEmprestimoAdapter :
    ListAdapter<EmprestimoGrupo, GrupoEmprestimoAdapter.ViewHolder>(DiffCallback()) {

    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): ViewHolder {
        val v = LayoutInflater.from(parent.context)
            .inflate(R.layout.item_grupo_emprestimo_expandable, parent, false)
        return ViewHolder(v)
    }

    override fun onBindViewHolder(holder: ViewHolder, position: Int) {
        holder.bind(getItem(position))
    }

    class ViewHolder(itemView: View) : RecyclerView.ViewHolder(itemView) {
        private val ivCapaLivro: ImageView = itemView.findViewById(R.id.ivCapaLivro)
        private val tvTituloLivro: TextView = itemView.findViewById(R.id.tvTituloLivro)
        private val tvQtdEmprestimos: TextView = itemView.findViewById(R.id.tvQtdEmprestimos)
        private val layoutDetalhes: LinearLayout = itemView.findViewById(R.id.layoutDetalhes)

        fun bind(grupo: EmprestimoGrupo) {
            tvTituloLivro.text = grupo.tituloLivro
            tvQtdEmprestimos.text = "${grupo.emprestimos.size} empréstimos realizados"

            val capa = grupo.capaLivro
            if (!capa.isNullOrBlank()) {
                Glide.with(itemView.context)
                    .load(if (capa.startsWith("//")) "https:$capa" else capa)
                    .placeholder(R.drawable.ic_capa_placeholder)
                    .error(R.drawable.ic_capa_placeholder)
                    .into(ivCapaLivro)
            } else {
                ivCapaLivro.setImageResource(R.drawable.ic_capa_placeholder)
            }

            layoutDetalhes.visibility = View.GONE

            itemView.setOnClickListener {
                if (layoutDetalhes.visibility == View.VISIBLE) {
                    layoutDetalhes.visibility = View.GONE
                } else {
                    layoutDetalhes.visibility = View.VISIBLE
                    preencherDetalhes(grupo)
                }
            }
        }

        private fun preencherDetalhes(grupo: EmprestimoGrupo) {
            layoutDetalhes.removeAllViews()
            val inflater = LayoutInflater.from(itemView.context)

            grupo.emprestimos.forEach { e ->
                val linha = inflater.inflate(R.layout.item_detalhe_emprestimo, layoutDetalhes, false)
                val tvPessoa = linha.findViewById<TextView>(R.id.tvPessoaDetalhe)
                val tvInfo = linha.findViewById<TextView>(R.id.tvInfoDetalhe)
                val tvStatus = linha.findViewById<TextView>(R.id.tvStatusDetalhe)

                val telefone = e.telefone ?: "-"
                val previsao = e.dataPrevistaDevolucao ?: "Sem prazo"

                // Determina status
                val status = when {
                    e.devolvido -> "Devolvido"
                    e.dataPrevistaDevolucao != null -> {
                        val sdf = SimpleDateFormat("yyyy-MM-dd", Locale.getDefault())
                        val dataPrevista = try { sdf.parse(e.dataPrevistaDevolucao) } catch (_: Exception) { null }
                        if (dataPrevista != null && dataPrevista.before(Date())) "Atrasado" else "Em aberto"
                    }
                    else -> "Em aberto"
                }

                tvPessoa.text = e.pessoa
                tvInfo.text = "Tel: $telefone | Emp.: ${e.dataEmprestimo} | Prev.: $previsao"

                // Aplica cor dinâmica ao chip de status
                val context = itemView.context
                val cor = when (status) {
                    "Devolvido" -> ContextCompat.getColor(context, android.R.color.holo_green_dark)
                    "Atrasado" -> ContextCompat.getColor(context, android.R.color.holo_red_dark)
                    else -> ContextCompat.getColor(context, android.R.color.holo_orange_dark)
                }
                tvStatus.text = status
                tvStatus.setTextColor(cor)

                layoutDetalhes.addView(linha)
            }
        }
    }

    class DiffCallback : DiffUtil.ItemCallback<EmprestimoGrupo>() {
        override fun areItemsTheSame(o: EmprestimoGrupo, n: EmprestimoGrupo) = o.livroId == n.livroId
        override fun areContentsTheSame(o: EmprestimoGrupo, n: EmprestimoGrupo) = o == n
    }
}
