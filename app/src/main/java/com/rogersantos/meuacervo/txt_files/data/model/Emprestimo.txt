package com.rogersantos.meuacervo.data.model

import androidx.room.Entity
import androidx.room.ForeignKey
import androidx.room.Index
import androidx.room.PrimaryKey

@Entity(
    tableName = "emprestimos",
    foreignKeys = [
        ForeignKey(
            entity = Livro::class,                // entidade pai
            parentColumns = ["id"],               // coluna da tabela Livro
            childColumns = ["livroId"],           // coluna da tabela Emprestimo
            onDelete = ForeignKey.CASCADE         // exclusão em cascata
        )
    ],
    indices = [Index("livroId")]                  // índice para melhorar consultas
)
data class Emprestimo(
    @PrimaryKey(autoGenerate = true) val id: Int = 0,
    val livroId: Int,                        // vínculo com Livro
    val pessoa: String,
    val telefone: String?,
    val dataEmprestimo: String,
    val dataPrevistaDevolucao: String?,
    var devolvido: Boolean = false,
    var dataDevolucao: String? = null,
    val lembreteWorkId: String? = null
)
